Calcular TRAMA (Media Móvil Adaptativa) // La fórmula ajusta el peso según la volatilidad reciente const trama = dataPoints.map((val, i) => { // Lógica de suavizado TRAMA aquí return val.close * 0.99; // Placeholder para el cálculo matemático }); // 2. Ejecutar estrategia UAlgo const lastPrice = dataPoints[dataPoints.length - 1].close; const lastTrama = trama[trama.length - 1]; if (lastPrice > lastTrama) { return { signal: "COMPRA", sl: lastPrice * 0.99, tp: lastPrice * 1.02 }; } else { return { signal: "VENTA", sl: lastPrice * 1.01, tp: lastPrice * 0.98 }; } }
