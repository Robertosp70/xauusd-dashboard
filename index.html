async function fetchMarketData() { const apiKey = 'R0112E15SLBWSGP8'; const symbol = 'XAUUSD'; const url = `https://www.alphavantage.co/query?function=GLOBAL_QUOTE&symbol=${symbol}&apikey=${apiKey}`; try { const response = await fetch(url); const data = await response.json(); // Verificación lógica: ¿Llegaron los datos? if (data['Global Quote'] && data['Global Quote']['05. price']) { const price = parseFloat(data['Global Quote']['05. price']); console.log("Precio obtenido:", price); // Esto lo verás en la consola (F12) return price; } else { console.error("Error: Estructura de datos inesperada o límite de API alcanzado."); } } catch (error) { console.error("Error de conexión:", error); } }
